#find the data with categorical variables
mydata2=titanic_train[,c("Survived","Pclass","Sex","SibSp","Parch","Embarked")]
f41=function(x)
 {uniqv=unique(x)
v=uniqv[which.max(tabulate(match(x,uniqv)))]
return (v)
}
# find the mode
df1=map_dfr(mydata2,f41)
f42=function(x)
 {uniqv=unique(x)
v=length(uniqv)
return (v)
}
